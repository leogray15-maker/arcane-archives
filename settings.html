// Profile picture upload handler document.getElementById('profilePictureInput').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; if (file.size > 5 * 1024 * 1024) { alert('❌ Image too large. Please choose an image under 5MB.'); return; } if (!file.type.startsWith('image/')) { alert('❌ Please select an image file.'); return; } const reader = new FileReader(); reader.onload = function(event) { const imageData = event.target.result; localStorage.setItem('customAvatarUrl', imageData); const preview = document.getElementById('currentAvatarPreview'); preview.style.backgroundImage = `url(${imageData})`; preview.textContent = ''; document.getElementById('removeAvatarBtn').style.display = 'inline-block'; const navAvatar = document.getElementById('userAvatar'); if (navAvatar) { navAvatar.style.backgroundImage = `url(${imageData})`; navAvatar.style.backgroundSize = 'cover'; navAvatar.style.backgroundPosition = 'center'; navAvatar.textContent = ''; } alert('✅ Profile picture updated successfully!'); }; reader.readAsDataURL(file); }); window.removeCustomAvatar = function() { if (!confirm('Remove custom profile picture?')) return; localStorage.removeItem('customAvatarUrl'); const preview = document.getElementById('currentAvatarPreview'); const userEmail = auth.currentUser?.email || 'User'; const letter = localStorage.getItem('userAvatar') || userEmail.charAt(0).toUpperCase(); preview.style.backgroundImage = 'none'; preview.textContent = letter; document.getElementById('removeAvatarBtn').style.display = 'none'; const navAvatar = document.getElementById('userAvatar'); if (navAvatar) { navAvatar.style.backgroundImage = 'none'; navAvatar.textContent = letter; } alert('✅ Profile picture removed'); }; // Load custom avatar on page load window.addEventListener('load', () => { const customAvatar = localStorage.getItem('customAvatarUrl'); const preview = document.getElementById('currentAvatarPreview'); if (customAvatar && preview) { preview.style.backgroundImage = `url(${customAvatar})`; preview.textContent = ''; const removeBtn = document.getElementById('removeAvatarBtn'); if (removeBtn) removeBtn.style.display = 'inline-block'; } });
